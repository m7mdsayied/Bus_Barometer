% 1. Page Layout
\usepackage[margin=0in]{geometry}

\usepackage{fontspec}

% Set Default Language to Arabic
\setmainlanguage{arabic}
\setotherlanguage{english}

\setmainfont[
    Path = fonts/,
    UprightFont = Almarai-Regular.ttf,
    BoldFont = Almarai-Bold.ttf,
    Script = Arabic

\newfontfamily\arabicfont[
    Path = fonts/,
    UprightFont = Almarai-Regular.ttf,
    BoldFont = Almarai-Bold.ttf,
    Script = Arabic
]{Almarai}

% 5. Hyperlinks
\usepackage[hidelinks]{hyperref}

% 8. Paragraph Formatting for Arabic
\setlength{\parindent}{0pt} % Often preferred in modern Arabic reports
\linespread{1.3}            % Arabic fonts usually need more line height


\usepackage{catchfile}  % For reading override files

\newcommand{\ECESContent}[2]{%
  \IfFileExists{overrides/#1.tex}{%
    \input{overrides/#1.tex}%
  }{%
    #2%
  }%
}

% 9. Automatic Backgrounds & Page Numbering
\usepackage{eso-pic}

% Create a variable to store the current background image filename
\def\CurrentBgImage{con_bg_ar.png} % Default background

% Create a switch to easily turn page numbers on/off (e.g., off for cover page)
\newif\ifShowPageNum
\ShowPageNumtrue

% Command to change the background image easily in your document
\newcommand{\SetBgImage}[1]{
    \def\CurrentBgImage{#1}
}

% Set up the universal background to automatically apply to EVERY page
\AddToShipoutPictureBG{%
    \begin{tikzpicture}[remember picture, overlay]
        % 1. Draw the Background Image dynamically
        \node[anchor=north west, inner sep=0pt] at (current page.north west) {
            \includegraphics[width=\paperwidth, height=\paperheight]{\CurrentBgImage}
        };
        
        % 2. Draw the Page Number (only if \ShowPageNum is true)
        \ifShowPageNum
            \node[anchor=south west] at ([xshift=1cm, yshift=1.2cm]current page.south west) {
                \fontsize{14}{14}\selectfont \textbf{\thepage}
            };
        \fi
    \end{tikzpicture}
}
