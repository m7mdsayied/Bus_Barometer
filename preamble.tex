% 9. Automatic Backgrounds & Page Numbering
\usepackage{eso-pic}

% Create a variable to store the current background image filename
\def\CurrentBgImage{con_bg.png} % Default background

% Create a switch to easily turn page numbers on/off (e.g., off for cover page)
\newif\ifShowPageNum
\ShowPageNumtrue

% Command to change the background image easily in your document
\newcommand{\SetBgImage}[1]{
    \def\CurrentBgImage{#1}
}

% Set up the universal background to automatically apply to EVERY page
\AddToShipoutPictureBG{%
    \begin{tikzpicture}[remember picture, overlay]
        % 1. Draw the Background Image dynamically
        \node[anchor=north west, inner sep=0pt] at (current page.north west) {
            \includegraphics[width=\paperwidth, height=\paperheight]{\CurrentBgImage}
        };
        
        % 2. Draw the Page Number (only if \ShowPageNum is true)
        \ifShowPageNum
            \node[anchor=south west] at ([xshift=1cm, yshift=1.2cm]current page.south west) {
                \fontsize{14}{14}\selectfont \textbf{\thepage}
            };
        \fi
    \end{tikzpicture}
}
